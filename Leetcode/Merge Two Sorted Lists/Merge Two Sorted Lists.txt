Q. Merge Two Sorted Lists

You are given the heads of two sorted linked lists list1 and list2.

Merge the two lists into one sorted list. The list should be made by splicing together the nodes of the first two lists.

Return the head of the merged linked list.

 

Example 1:


Input: list1 = [1,2,4], list2 = [1,3,4]
Output: [1,1,2,3,4,4]
Example 2:

Input: list1 = [], list2 = []
Output: []
Example 3:

Input: list1 = [], list2 = [0]
Output: [0]
 

Constraints:

The number of nodes in both lists is in the range [0, 50].
-100 <= Node.val <= 100
Both list1 and list2 are sorted in non-decreasing order.

A. 

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {

        //if the lists are empty return the other list

        if(list1==NULL)return list2;
        if(list2==NULL)return list1;

        //if the value of 1st is greater swap it

        if(list1->val>list2->val)
        swap(list1,list2);

        // Create node to store result

        ListNode *res = list1;

        while(list1!=NULL && list2!=NULL)
        {

        // use this variable for swapping

         ListNode* temp=NULL;

        //  check if list 1 is null and the value is less than value 2

         while(list1!=NULL && list1->val <= list2->val)
         {
             temp=list1;
             list1=list1->next;
         }
         temp->next = list2;
         swap(list1,list2);
        }    
        return res;
    }
};
